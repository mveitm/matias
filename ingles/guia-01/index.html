<script>

    const evaluationCriteria = [

        { kw: ["progreso", "civilización", "cultura", "libertad"], fb: "Correcto. Bello destaca la universidad como pilar de la civilización." },

        { kw: ["conexión", "interdisciplinar", "conocimiento", "unión"], fb: "Bien. Se refiere a la unidad orgánica del saber humano." },

        { kw: ["identidad", "vínculo", "patrimonio", "comunicación"], fb: "Así es. La lengua es el lazo cultural de la nación." },

        { kw: ["ilustración", "pensamiento", "pueblo", "independencia"], fb: "Correcto. La ciencia garantiza una libertad real, no solo nominal." },

        { kw: ["equilibrio", "necesarias", "espíritu", "humana"], fb: "Excelente. Las letras cultivan la moral y el pensamiento crítico." },

        { kw: ["dogma", "política", "independencia", "autonomía"], fb: "Muy bien. Defiende la objetividad frente a facciones políticas." }

    ];

    function evaluateQuiz() {

        let score = 0;

        const feedbackList = document.getElementById('feedback-list');

        feedbackList.innerHTML = '';

       

        for (let i = 1; i <= 6; i++) {

            const answer = document.getElementById(`q${i}`).value.toLowerCase();

            const criteria = evaluationCriteria[i-1];

            const matched = criteria.kw.filter(word => answer.includes(word));

           

            let points = matched.length >= 2 ? 1 : (matched.length === 1 ? 0.5 : 0);

            score += points;

            const div = document.createElement('div');

            div.className = `feedback-item ${points >= 0.5 ? 'correct' : 'incorrect'}`;

            div.innerHTML = `<strong>P${i}:</strong> ${points === 1 ? criteria.fb : 'Incompleta. Faltaron conceptos como: ' + criteria.kw.slice(0,2).join(', ')}<br><small>Tu respuesta: ${answer}</small>`;

            feedbackList.appendChild(div);

        }

        document.getElementById('total-score').innerHTML = `Aprobación: ${Math.round((score/6)*100)}%`;

        document.getElementById('results').style.display = 'block';

        window.scrollTo(0, document.body.scrollHeight);

    }

    function downloadTXT() {

        let content = "RESULTADOS DE EVALUACIÓN - ANDRÉS BELLO\n";

        content += document.getElementById('total-score').innerText + "\n\n";

       

        for (let i = 1; i <= 6; i++) {

            content += `Pregunta ${i}: ${document.getElementById(`q${i}`).value}\n`;

        }

       

        const blob = new Blob([content], { type: 'text/plain' });

        const a = document.createElement('a');

        a.href = URL.createObjectURL(blob);

        a.download = 'resultados_evaluacion.txt';

        a.click();

    }

</script>
